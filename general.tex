    % #1 date #2 error times #3 index
    \DeclareDocumentCommand { \addto } { O{#2} D(){1} m }
      {
        \sdIfReviewT{default}{##1}{\mydate}
          {
            \def\mysave####1{
              \sdsave*{####1}{part}{#3}
              \sdsave*{####1}{index}{##3}
              \ifnum##2>1\sdsave*{####1}{error}{##2}
                \sdsave*{####1}{color}{\protect\mylabel{error}}
              \else\sdsave*{####1}{error}{}
                \sdsave*{####1}{color}{\protect\mylabel{label}}
              \fi
            }
            \IfBooleanTF { #1 }
              { \mysave{single} }
              { \mysave{multi} }
          }
      }
    #4
  }

  {

      \textsubscript{%

  }
}
